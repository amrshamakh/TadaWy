--q5
select makes.make, numberofcar=count(*)  from VehicleMasterDetails
inner join Makes on
VehicleMasterDetails.Make=Makes.Make
where VehicleMasterDetails.Year between 1950 and 2000 
group by Makes.make having count(*)>12000
order by numberofcar desc;


--q7
select * , cast(numberofcar as float) / cast(totalcars as float) from (
select makes.make, numberofcar=count(*),(select count(*) from VehicleMasterDetails)as totalcars

from VehicleMasterDetails
inner join Makes on
VehicleMasterDetails.Make=Makes.Make
where (VehicleMasterDetails.Year between 1950 and 2000 )
group by Makes.make) R1
order by numberofcar desc;

--q8
select makes.make,FuelTypes.FuelTypeName, numberofcar=count(*)  from VehicleMasterDetails
inner join Makes on
VehicleMasterDetails.Make=Makes.Make
inner join FuelTypes on
VehicleMasterDetails.FuelTypeName=FuelTypes.FuelTypeName
where VehicleMasterDetails.Year between 1950 and 2000 
group by Makes.Make,FuelTypes.FuelTypeName
order by Makes.make;


--qu10
select Makes.Make,FuelTypes.FuelTypeName 
from VehicleDetails inner join Makes
on VehicleDetails.MakeID=Makes.MakeID
inner join FuelTypes
on VehicleDetails.FuelTypeID=FuelTypes.FuelTypeID
where FuelTypeName='gas'
group by Makes.Make,FuelTypes.FuelTypeName;
